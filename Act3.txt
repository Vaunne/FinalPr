<script>

function handlecourseDropdownChange() {
const courseSelection = document.getElementById('courseDropdown').value;

if (courseSelection === 'BSIT') {
document.getElementById('Sections1Dropdown').style.display = 'block';
document.getElementById('Sections2Dropdown').style.display = 'none';
document.getElementById('Sections3Dropdown').style.display = 'none';
document.getElementById('Sections4Dropdown').style.display = 'none';
} else if (courseSelection === 'BSCS') {
document.getElementById('Sections1Dropdown').style.display = 'none';
document.getElementById('Sections2Dropdown').style.display = 'block';
document.getElementById('Sections3Dropdown').style.display = 'none';
document.getElementById('Sections4Dropdown').style.display = 'none';
} else if (courseSelection === 'BSDA') {
document.getElementById('Sections1Dropdown').style.display = 'none';
document.getElementById('Sections2Dropdown').style.display = 'none';
document.getElementById('Sections3Dropdown').style.display = 'block';
document.getElementById('Sections4Dropdown').style.display = 'none';
}else if (courseSelection === 'BSMMA') {
document.getElementById('Sections1Dropdown').style.display = 'none';
document.getElementById('Sections2Dropdown').style.display = 'none';
document.getElementById('Sections3Dropdown').style.display = 'none';
document.getElementById('Sections4Dropdown').style.display = 'block';
}else {
document.getElementById('Sections1Dropdown').style.display = 'none';
document.getElementById('Sections2Dropdown').style.display = 'none';
document.getElementById('Sections3Dropdown').style.display = 'none';
document.getElementById('Sections4Dropdown').style.display = 'none';
}
}

function saveRecord() {
var name = document.getElementById('name').value;
var course = document.getElementById('courseDropdown').value;
var section = document.getElementById('Sections1Dropdown').value;


var table = document.getElementById('recordTable');
var row = table.insertRow(-1);
var cell1 = row.insertCell(0);
var cell2 = row.insertCell(1);
var cell3 = row.insertCell(2);
var cell4 = row.insertCell(3);
cell1.innerHTML = name;
cell2.innerHTML = course;
cell3.innerHTML = section;
cell4.innerHTML = "<button onclick='deleteRow(this)'>DELETE</button>";
}

function deleteRow(row) {
var i = row.parentNode.parentNode.rowIndex;
document.getElementById('recordTable').deleteRow(i);
}
